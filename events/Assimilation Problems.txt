country_event = {

	id = 99857

	trigger = {
		any_owned_province = {
				is_core = EST
			}
		NOT = {
			exists = EST
			primary_culture = estonian
			has_global_flag = estonian_problem
		}
	}

	mean_time_to_happen =  {
		months = 3
	}
	
	title = "EVTNAME99851"
	desc = "EVTDESC99851"
	picture = "polrebel"

	option = {
		name = "EVTOPTA99851"
		set_global_flag = estonian_problem
		any_owned = {
			limit = {
				is_core = EST
			}
			add_province_modifier = {
						name = separatist
						duration = -1
			}
		}
		
	}
}
